# % of missing values

def missing_values(df):
    
    print ('% of Null Values:\n', (df.isnull().sum()/df.shape[0])*100)
    print('\nDF shape before dropping Null values:', df.shape)
    df = df.dropna()
    df.reset_index(drop=True, inplace=True)
    print('\nDF shape after dropping Null values:', df.shape)
    
    return df

# check duplicate records

def check_duplicates(df):
    if (df.shape[0] != df.drop_duplicates().shape[0]):
        print('No. of duplicate Records Present:', df.shape[0] - df.drop_duplicates().shape[0])
        print('\nDF shape before dropping duplicate values:', df.shape)
        df = df.drop_duplicates()
        df.reset_index(drop=True, inplace=True)
        print('\nDF shape after dropping duplicate values:', df.shape)
        
        return df
    else:
        print('No duplicate values present!')
        
        return df
    

# Drop columns which might be redundant

col_list = ['customerID']

def drop_red_cols(df, col_list=[]):
    print('Dropping columns:', col_list)
    print('\nDF shape before dropping redundant columns:', df.shape)
    df.drop(col_list, axis=1, inplace=True)
    print('\nDF shape after dropping redundant columns:', df.shape)
    
    return df
    

# Encode the categorical variables

from sklearn.preprocessing import LabelEncoder

def encode_cat_var(df):
    cat_var = {}
    for i in df.columns:
        if(df[i].dtype == "O"):
            print('Encoding "', i, '"...')
            le = LabelEncoder()
            df[i] = le.fit_transform(df[i])
            cat_var[i] = le
            
    return df

# -> Long tenure justifies that they are not going to stop the services because they are happy customers; same for contract 
# -> If customers are focusing on online security, that means they want to continue their services
# -> If customers are using techsupport that means they are interested in resolving the existing issues in order to continue the services smoothly

# Positively correlated (higher chances of churning):
# -> If monthly charges are high then chances of leaving are also high
# -> If customer is becoming aged then higher chances of leaving the services because they won't be moving around too much
